  def self.merge_rubygems
    in_install_dir do
      gems = Dir["*/lib/ruby/gems"]

      unless test ?d, "../gems" then
        FileUtils.mv gems.first, ".."
      end

      gems.each do |d|
        FileUtils.rm_rf d
        FileUtils.ln_sf "../../../../gems", d
      end
    end
  end
